tmux                    # Start a new tmux session

ðŸ”¹ DETACHING & EXITING
----------------------
Ctrl-b d                # Detach from current session
exit                    # Exit current pane or session

ðŸ”¹ SPLITTING WINDOWS
--------------------
Ctrl-b "                # Split pane horizontally
Ctrl-b %                # Split pane vertically

ðŸ”¹ MOVING BETWEEN PANES
------------------------
Ctrl-b o                # Cycle through panes
Ctrl-b ;                # Go to last active pane
Ctrl-b q                # Show pane numbers briefly

ðŸ”¹ RESIZING PANES
-----------------
Ctrl-b :resize-pane -L 10    # Resize left
Ctrl-b :resize-pane -R 10    # Resize right
Ctrl-b :resize-pane -U 5     # Resize up
Ctrl-b :resize-pane -D 5     # Resize down

ðŸ”¹ WORKING WITH WINDOWS (TABS)
------------------------------
Ctrl-b c                # Create new window
Ctrl-b ,                # Rename window
Ctrl-b n / p            # Next / Previous window
Ctrl-b w                # List all windows
Ctrl-b &                # Kill current window
